# Sprints 2-3-4 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement 5 features across Sprints 2, 3, 4: Note Google badge, Formulaire de contact, Page Financement, Vue calendrier sessions, Transitions de page (Framer Motion).

**Architecture:** Server Components fetch data from Supabase and pass props to Client Components. Server Actions handle all mutations with `revalidatePath`. Two new Supabase tables (`google_rating`, `contact_messages`), two static pages (`/financement`, `/calendrier`), and Framer Motion page transitions wired into the root layout.

**Tech Stack:** Next.js 16 App Router, TypeScript strict, Tailwind v4, Supabase (`createAdminClient` via `@supabase/supabase-js`), Resend + React Email, Framer Motion v12 (already installed).

---

## Codebase conventions to follow

- **Supabase admin client:** always `createAdminClient()` from `@/lib/supabase/server` for server actions and admin pages. Never use `createClient()` for mutations.
- **Auth guard:** admin pages call `requireAuth()` (already in `src/app/admin/actions.ts`). Replicate the pattern in new server action files.
- **Server Actions:** `'use server'` at top of file, return `{ error: string }` on failure, call `revalidatePath(...)` on success.
- **Email:** import from `@react-email/components`, send with Resend via `new Resend(process.env.RESEND_API_KEY)`.
- **Admin UI:** sharp corners (no `rounded-*`), `bg-surface-container-lowest border border-surface-container-highest` cards, `font-label-caps tracking-widest uppercase` buttons in `bg-primary text-on-primary`.
- **Sessions table fields:** `id`, `formation_id`, `date_debut` (date), `date_fin` (date), `places_disponibles` (int). No `heure_debut`/`heure_fin` columns.

```sql
-- ─── google_rating ─────────────────────────────────────────────────────────
create table if not exists google_rating (
  id int primary key default 1 check (id = 1),
  note decimal(2,1) not null default 5.0,
  nb_avis int not null default 0,
  google_url text not null default '',
  updated_at timestamptz not null default now()
);

alter table google_rating enable row level security;

create policy "Public read google_rating"
  on google_rating for select using (true);

create policy "Service role all google_rating"
  on google_rating for all using (auth.role() = 'service_role');

-- seed the single row so upsert works immediately
insert into google_rating (id, note, nb_avis, google_url)
values (1, 5.0, 0, '')
on conflict (id) do nothing;

-- ─── contact_messages ───────────────────────────────────────────────────────
create table if not exists contact_messages (
  id uuid primary key default gen_random_uuid(),
  prenom text not null,
  nom text not null,
  email text not null,
  message text not null,
  lu boolean not null default false,
  created_at timestamptz not null default now()
);

alter table contact_messages enable row level security;

create policy "Public insert contact_messages"
  on contact_messages for insert with check (true);

create policy "Service role all contact_messages"
  on contact_messages for all using (auth.role() = 'service_role');
```

```tsx
export default async function GoogleRatingBadge() {
  let rating: { note: number; nb_avis: number; google_url: string } | null = null

  try {

      .from('google_rating')
```

```ts
// src/app/admin/(protected)/google-rating/actions.ts
'use server'

import { createClient } from '@/lib/supabase/server'
import { createAdminClient } from '@/lib/supabase/server'
import { redirect } from 'next/navigation'
import { revalidatePath } from 'next/cache'

async function requireAuth() {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) redirect('/admin/login')
}

export async function updateGoogleRating(fd: FormData) {
  await requireAuth()

  const noteStr = (fd.get('note') as string | null)?.trim() ?? ''
  const nbAvisStr = (fd.get('nb_avis') as string | null)?.trim() ?? ''
  const googleUrl = (fd.get('google_url') as string | null)?.trim() ?? ''

  const note = parseFloat(noteStr)
  const nb_avis = parseInt(nbAvisStr, 10)

  if (isNaN(note) || note < 0 || note > 5) return { error: 'Note invalide (0–5)' }
  if (isNaN(nb_avis) || nb_avis < 0) return { error: 'Nombre d\'avis invalide' }

  const supabase = await createAdminClient()
  const { error } = await supabase
    .from('google_rating')
    .upsert([{ id: 1, note, nb_avis, google_url: googleUrl, updated_at: new Date().toISOString() }], { onConflict: 'id' })

  if (error) return { error: 'Erreur lors de la sauvegarde : ' + error.message }

  revalidatePath('/')
  revalidatePath('/admin/google-rating')
  return { success: true }
}
```

```tsx
'use client'

import { useState, useTransition } from 'react'
import { updateGoogleRating } from './actions'

interface Props {
  current: { note: number; nb_avis: number; google_url: string }
}

export default function GoogleRatingForm({ current }: Props) {
  const [note, setNote] = useState(String(current.note))
  const [nbAvis, setNbAvis] = useState(String(current.nb_avis))
  const [url, setUrl] = useState(current.google_url)
  const [status, setStatus] = useState<'idle' | 'saving' | 'saved' | 'error'>('idle')
  const [errorMsg, setErrorMsg] = useState('')
  const [isPending, startTransition] = useTransition()

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    setStatus('saving')
    const fd = new FormData()
    fd.set('note', note)
    fd.set('nb_avis', nbAvis)
    fd.set('google_url', url)

    startTransition(async () => {
      const result = await updateGoogleRating(fd)
      if (result?.error) {
        setErrorMsg(result.error)
        setStatus('error')
        setTimeout(() => setStatus('idle'), 4000)
      } else {
        setStatus('saved')
        setTimeout(() => setStatus('idle'), 3000)
      }
    })
  }

  const inputClass = "w-full border border-surface-container-highest bg-surface-container-lowest px-4 py-2.5 text-sm text-on-surface focus:outline-none focus:border-primary"

  return (
    <form onSubmit={handleSubmit} className="space-y-6 max-w-lg">
      <div>
        <label htmlFor="note" className="block text-sm font-medium text-on-surface mb-1">
          Note (ex : 4.9)
        </label>
        <input
          id="note"
          type="number"
          step="0.1"
          min="0"
          max="5"
          value={note}
          onChange={(e) => setNote(e.target.value)}
          className={inputClass}
          required
        />
      </div>

      <div>
        <label htmlFor="nb_avis" className="block text-sm font-medium text-on-surface mb-1">
          Nombre d'avis
        </label>
        <input
          id="nb_avis"
          type="number"
          min="0"
          value={nbAvis}
          onChange={(e) => setNbAvis(e.target.value)}
          className={inputClass}
          required
        />
      </div>

      <div>
        <label htmlFor="google_url" className="block text-sm font-medium text-on-surface mb-1">
          URL Google Maps (lien vers votre fiche)
        </label>
        <input
          id="google_url"
          type="url"
          value={url}
          onChange={(e) => setUrl(e.target.value)}
          placeholder="https://maps.google.com/..."
          className={inputClass}
        />
      </div>

      <div className="flex items-center justify-between">
        <div className="text-xs">
          {status === 'saved' && <span className="text-green-700">✓ Enregistré</span>}
          {status === 'error' && <span className="text-red-600">{errorMsg}</span>}

          type="submit"

        >
          {status === 'saving' ? 'Enregistrement…' : 'Enregistrer'}
        </button>
      </div>
    </form>
  )
}
```

```ts
// src/app/contact/actions.ts
'use server'

import { createAdminClient } from '@/lib/supabase/server'
import { Resend } from 'resend'
import ContactNotificationEmail from '@/emails/ContactNotificationEmail'
import ContactConfirmationEmail from '@/emails/ContactConfirmationEmail'
import React from 'react'

const resend = new Resend(process.env.RESEND_API_KEY)

export async function submitContact(fd: FormData) {
  // Honeypot check — bots fill the hidden 'website' field
  const honeypot = (fd.get('website') as string | null) ?? ''
  if (honeypot.trim() !== '') return { success: true } // silently ignore

  const prenom  = (fd.get('prenom')  as string | null)?.trim() ?? ''
  const nom     = (fd.get('nom')     as string | null)?.trim() ?? ''
  const email   = (fd.get('email')   as string | null)?.trim() ?? ''
  const message = (fd.get('message') as string | null)?.trim() ?? ''

  if (prenom.length < 2) return { error: 'Prénom trop court.' }
  if (nom.length < 2)    return { error: 'Nom trop court.' }
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) return { error: 'Adresse email invalide.' }
  if (message.length < 10) return { error: 'Message trop court (minimum 10 caractères).' }
  if (message.length > 2000) return { error: 'Message trop long (maximum 2000 caractères).' }

  const supabase = await createAdminClient()

  const { error: dbError } = await supabase
    .from('contact_messages')
    .insert([{ prenom, nom, email, message }])

  if (dbError) return { error: 'Erreur lors de l\'envoi. Veuillez réessayer.' }

  const ownerEmail = process.env.CONTACT_EMAIL ?? 'beautyhomeconcept@gmail.com'

  // Notification to owner
  await resend.emails.send({
    from: 'Beauty Home Concept <contact@beautyhomeconcept.fr>',
    to: ownerEmail,
    subject: `Nouveau message de ${prenom} ${nom}`,
    react: React.createElement(ContactNotificationEmail, { prenom, nom, email, message }),
  })

  // Confirmation to visitor
  await resend.emails.send({
    from: 'Beauty Home Concept <contact@beautyhomeconcept.fr>',
    to: email,
    subject: 'Votre message a bien été reçu',
    react: React.createElement(ContactConfirmationEmail, { prenom, message }),
  })

  return { success: true }
}
```

```tsx
'use client'

import { useState } from 'react'
import Link from 'next/link'

interface Session {
  id: string
  date_debut: string
  date_fin: string
  places_disponibles: number
  formation_id: string
  formation_titre: string
  formation_prix: number
}

interface Props {
  sessions: Session[]
  formations: { id: string; titre: string }[]
}

function formatDateRange(debut: string, fin: string): string {
  const d1 = new Date(debut)
  const d2 = new Date(fin)
  const opts: Intl.DateTimeFormatOptions = { weekday: 'long', day: 'numeric', month: 'long', year: 'numeric' }
  if (debut === fin) return d1.toLocaleDateString('fr-FR', opts)
  return `${d1.toLocaleDateString('fr-FR', { weekday: 'long', day: 'numeric', month: 'long' })} – ${d2.toLocaleDateString('fr-FR', opts)}`
}

function getMonthLabel(dateStr: string): string {
  const d = new Date(dateStr)
  return d.toLocaleDateString('fr-FR', { month: 'long', year: 'numeric' })
}

function PlacesBadge({ places }: { places: number }) {
  if (places <= 0) return <span className="text-[10px] font-label-caps tracking-wider uppercase text-red-600 bg-red-50 px-2 py-0.5 border border-red-200">Complet</span>
  if (places <= 2)  return <span className="text-[10px] font-label-caps tracking-wider uppercase text-amber-700 bg-amber-50 px-2 py-0.5 border border-amber-200">{places} place{places > 1 ? 's' : ''}</span>
  return <span className="text-[10px] font-label-caps tracking-wider uppercase text-on-surface-variant bg-surface-container px-2 py-0.5 border border-outline-variant/30">{places} places</span>
}

export default function CalendrierView({ sessions, formations }: Props) {
  const [filterFormationId, setFilterFormationId] = useState<string>('all')

  const filtered = filterFormationId === 'all'
    ? sessions
    : sessions.filter((s) => s.formation_id === filterFormationId)

  // Group by month
  const groups: { label: string; sessions: Session[] }[] = []
  for (const s of filtered) {
    const label = getMonthLabel(s.date_debut)
    const existing = groups.find((g) => g.label === label)
    if (existing) {
      existing.sessions.push(s)
    } else {
      groups.push({ label, sessions: [s] })
    }
  }

  return (
    <div>
      {/* Filter */}
      {formations.length > 1 && (
        <div className="mb-10">
          <select
            value={filterFormationId}
            onChange={(e) => setFilterFormationId(e.target.value)}
            className="border border-outline-variant/50 bg-surface-container-lowest px-4 py-2.5 text-sm text-on-surface focus:outline-none focus:border-primary"
            aria-label="Filtrer par formation"
          >
            <option value="all">Toutes les formations</option>
            {formations.map((f) => (
              <option key={f.id} value={f.id}>{f.titre}</option>
            ))}
          </select>
        </div>
      )}

      {filtered.length === 0 && (
        <p className="text-on-surface-variant text-sm italic py-12 text-center">
          Aucune session disponible pour le moment.
        </p>
      )}

      <div className="space-y-12">
        {groups.map((group) => (
          <div key={group.label}>
            <div className="font-label-caps text-[10px] tracking-widest text-primary uppercase pb-3 mb-4 border-b border-outline-variant/30">
              {group.label}
            </div>
            <div className="space-y-2">
              {group.sessions.map((s) => (
                <div
                  key={s.id}
                  className="bg-surface border border-surface-container-highest p-5 flex flex-col sm:flex-row sm:items-center sm:justify-between gap-4"
                >
                  <div className="space-y-1.5">
                    <p className="font-medium text-on-surface text-sm">
                      {s.formation_titre}
                    </p>
                    <p className="text-xs text-on-surface-variant capitalize">
                      {formatDateRange(s.date_debut, s.date_fin)}
                    </p>
                    <div className="flex items-center gap-3">
                      <PlacesBadge places={s.places_disponibles} />
                      <span className="text-xs text-on-surface-variant">
                        {s.formation_prix.toLocaleString('fr-FR')} €
                      </span>
                    </div>

                    className="flex-shrink-0 inline-block bg-primary text-on-primary px-5 py-2.5 text-[10px] font-label-caps tracking-widest uppercase hover:opacity-90 transition-opacity active:scale-[0.97] text-center"
                  >
                    {s.places_disponibles <= 0 ? 'Liste d\'attente' : 'Voir & réserver'}
                  </Link>
                </div>
              ))}
            </div>
          </div>
```

```tsx
  const { data: rawSessions } = await supabase
    .from('sessions')
    .select(`
      id,
      date_debut,
      date_fin,
      places_disponibles,
      formation_id,
      formations!inner (
        id,
        titre,
        prix
      )
    `)
    .gte('date_debut', today)
    .order('date_debut', { ascending: true })

  const sessions = (rawSessions ?? []).map((s: any) => ({
    id: s.id,
    date_debut: s.date_debut,
    date_fin: s.date_fin,
    places_disponibles: s.places_disponibles,
    formation_id: s.formation_id,
    formation_titre: s.formations.titre,
    formation_prix: s.formations.prix,
  }))

  // Deduplicate formations for filter dropdown
  const seen = new Set<string>()
  const formations: { id: string; titre: string }[] = []
  for (const s of sessions) {
    if (!seen.has(s.formation_id)) {
      seen.add(s.formation_id)
      formations.push({ id: s.formation_id, titre: s.formation_titre })
    }
  }

  return (
    <>
      <Header />
      <main className="pt-[100px] pb-24 bg-surface-container-lowest min-h-screen">
        <div className="max-w-container-max mx-auto px-margin-mobile md:px-margin-desktop">

          <Link

            className="inline-flex items-center gap-2 min-h-[44px] text-on-surface-variant hover:text-primary transition-colors font-label-caps text-[10px] uppercase tracking-widest mb-12"
          >

              className="material-symbols-outlined text-[16px]"
              style={{ fontVariationSettings: "'FILL' 0, 'wght' 300, 'GRAD' 0, 'opsz' 24" }}
            >
              arrow_back
            </span>
            Retour à l'accueil
          </Link>

          <div className="max-w-2xl mb-16">
            <span className="font-label-caps text-primary tracking-widest text-[10px] uppercase block mb-4">
              Planning
            </span>
            <h1 className="font-playfair text-4xl md:text-5xl text-on-surface mb-4 leading-tight">
              Sessions à venir
            </h1>
            <p className="text-on-surface-variant text-base leading-relaxed">
              Toutes les dates disponibles, mises à jour en temps réel.
            </p>
          </div>

          <div className="max-w-2xl">
            <CalendrierView sessions={sessions} formations={formations} />
          </div>

        </div>
      </main>
      <Footer />
    </>
  )
}
```

```tsx
// src/components/PageTransition.tsx
'use client'

import { motion, AnimatePresence, useReducedMotion } from 'framer-motion'
import { usePathname } from 'next/navigation'

export default function PageTransition({ children }: { children: React.ReactNode }) {
  const pathname = usePathname()
  const shouldReduceMotion = useReducedMotion()

  const variants = {
    hidden: { opacity: 0, y: shouldReduceMotion ? 0 : 8 },
    enter: { opacity: 1, y: 0 },
    exit: { opacity: 0, y: shouldReduceMotion ? 0 : -8 },
  }

  return (
    <AnimatePresence mode="wait">
      <motion.div
        key={pathname}
        initial="hidden"
        animate="enter"
        exit="exit"
        variants={variants}
        transition={
          shouldReduceMotion
            ? { duration: 0 }
            : {
                duration: 0.28,
                ease: [0.23, 1, 0.32, 1],
              }
        }
      >
        {children}
      </motion.div>
    </AnimatePresence>
  )
}
```
